            buttonActive = tabBtns[firstIndex]
            contentActive = tabContents[firstIndex]
 
            tabBtns[firstIndex].classList.add("is-active")

            Array.prototype.forEach.call(tabBtns , function (value,  key  ){

                    buttonActive = this;

                    _this.activate( this, "is-active")

                })
            });
 
            console.log(tabBtns)

    },
